! Static Routing Lab

! Router R1
enable
configure terminal

hostname R1

interface gigabitEthernet 0/0
 ip address 192.168.1.1 255.255.255.0
 no shutdown

interface gigabitEthernet 0/1
 ip address 10.0.0.1 255.255.255.252
 no shutdown

ip route 192.168.2.0 255.255.255.0 10.0.0.2

end
write memory


! Router R2
